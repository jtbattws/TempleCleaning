        // Mock Application Database State

        let state = {

                { id: 3, title: "Saturday Midday — Windows & High Fixtures", date: "August 15, 2026", time: "11:00 AM - 2:00 PM", capacity: 20, filled: 0 }

            ],

            volunteers: [

                { name: "John Burt", email: "john.burt@example.com", phone: "801-555-1234", shiftId: 1, type: "both" },

                { name: "Sarah Jenkins", email: "sarah.j@example.com", phone: "801-555-5678", shiftId: 2, type: "email" },

                { name: "David Miller", email: "d.miller@example.com", phone: "801-555-9012", shiftId: 2, type: "sms" }

            ],

            selectedShiftId: null

        };


        const ADMIN_PASSCODE = "1234"; // Simple, functional demo pass code


        // Initial render setup

        document.addEventListener("DOMContentLoaded", () => {

            renderShifts();

        });


        // Tab Switch Management UI

        function switchTab(target) {

            ['signup', 'checklist', 'admin'].forEach(tab => {

                const button = document.getElementById(`tab-${tab}`);

                const content = document.getElementById(`content-${tab}`);

                if (tab === target) {

                    button.classList.add('text-cyan-600', 'border-cyan-600');

                    button.classList.remove('text-slate-500', 'border-transparent');

                    content.classList.remove('hidden');

                } else {

                    button.classList.remove('text-cyan-600', 'border-cyan-600');

                    button.classList.add('text-slate-500', 'border-transparent');

                    content.classList.add('hidden');

                }

            });

        }


        // Render Shifts List View

        function renderShifts() {

            const container = document.getElementById('shifts-container');

            container.innerHTML = '';


            state.shifts.forEach(shift => {

                const remaining = shift.capacity - shift.filled;

                const isFull = remaining <= 0;


                const card = document.createElement('div');

                card.className = `p-4 border rounded-xl flex flex-col md:flex-row md:items-center md:justify-between transition ${isFull ? 'bg-slate-50 border-slate-200' : 'bg-white border-slate-200 hover:border-cyan-300 hover:shadow-sm'}`;

                

                card.innerHTML = `

                    <div class="space-y-1">

                        <h4 class="font-bold text-slate-900">${shift.title}</h4>

                        <div class="text-sm text-slate-600 flex flex-wrap gap-x-4 gap-y-1">

                            <span><i class="fa-regular fa-calendar mr-1 text-slate-400"></i> ${shift.date}</span>

                            <span><i class="fa-regular fa-clock mr-1 text-slate-400"></i> ${shift.time}</span>

                        </div>

                        <div class="text-xs font-semibold ${isFull ? 'text-red-500' : remaining <= 5 ? 'text-amber-600' : 'text-slate-500'}">

                            ${isFull ? 'Shift Full' : `${remaining} spots remaining / ${shift.capacity} total`}

                        </div>

                    </div>

                    <div class="mt-3 md:mt-0">

                        <button onclick="openRegistration(${shift.id})" ${isFull ? 'disabled' : ''} class="w-full md:w-auto px-4 py-2 rounded-lg text-sm font-semibold transition ${isFull ? 'bg-slate-200 text-slate-400 cursor-not-allowed' : 'bg-cyan-50 hover:bg-cyan-100 text-cyan-700'}">

                            ${isFull ? 'Filled' : 'Select Shift'}

                        </button>

                    </div>

                `;

                container.appendChild(card);

            });

        }


        // Registration Panel Actions

        function openRegistration(shiftId) {

            state.selectedShiftId = shiftId;

            const shift = state.shifts.find(s => s.id === shiftId);

            document.getElementById('selected-shift-title').innerText = `${shift.title} (${shift.date})`;

            

            const formContainer = document.getElementById('registration-form-container');

            formContainer.classList.remove('hidden');

            formContainer.scrollIntoView({ behavior: 'smooth' });

        }


        function cancelRegistration() {

            state.selectedShiftId = null;

            document.getElementById('signup-form').reset();

            document.getElementById('registration-form-container').add('hidden');

        }


        function handleFormSubmit(event) {

            event.preventDefault();

            

            const name = document.getElementById('vol-name').value;

            const email = document.getElementById('vol-email').value;

            const phone = document.getElementById('vol-phone').value;

            const type = document.getElementById('vol-reminder').value;


            // Update registration state tracking

            state.volunteers.push({

                name, email, phone, shiftId: state.selectedShiftId, type

            });


            // Increment filled value on active target shift

            const shift = state.shifts.find(s => s.id === state.selectedShiftId);

            shift.filled++;


            // Reset view state

            alert(`Success! Thank you for signing up, ${name}. A reminder note has been queued up.`);

            document.getElementById('signup-form').reset();

            document.getElementById('registration-form-container').classList.add('hidden');

            

            renderShifts();

            renderRoster(); // syncs dashboard roster background content

        }


        // Administrative Access Controls

        function checkAdminAuth() {

            const pin = document.getElementById('admin-passcode').value;

            if (pin === ADMIN_PASSCODE) {

                document.getElementById('admin-auth').classList.add('hidden');

                document.getElementById('admin-dashboard').classList.remove('hidden');

                renderRoster();

            } else {

                document.getElementById('auth-error').classList.remove('hidden');

            }

        }


        // Build Admin Dashboard rows

        function renderRoster() {

            const tbody = document.getElementById('roster-table-body');

            tbody.innerHTML = '';


            if (state.volunteers.length === 0) {

                tbody.innerHTML = `<tr><td colspan="4" class="p-4 text-center text-slate-400">No active volunteer registrations yet.</td></tr>`;

                return;

            }


            state.volunteers.forEach(v => {

                const shift = state.shifts.find(s => s.id === v.shiftId);

                const tr = document.createElement('tr');

                tr.innerHTML = `

                    <td class="p-4 font-medium text-slate-900">${v.name}</td>

                    <td class="p-4 text-slate-500">

                        <div><i class="fa-regular fa-envelope text-xs mr-1"></i> ${v.email}</div>

                        <div><i class="fa-solid fa-phone text-xs mr-1"></i> ${v.phone}</div>

                    </td>

                    <td class="p-4">

                        <span class="inline-flex px-2 py-1 text-xs font-semibold bg-cyan-50 text-cyan-700 rounded-md">${shift ? shift.title : 'Unknown Shift'}</span>

                    </td>

                    <td class="p-4 text-center text-xs text-slate-500 capitalize">${v.type}</td>

                `;

                tbody.appendChild(tr);

            });

        }


        // Flat CSV Data Exporter (Simulated Excel Download Integration)

        function exportToCSV() {

            let csvContent = "data:text/csv;charset=utf-8,Full Name,Email,Phone,Assigned Shift,Preference\n";

            

            state.volunteers.forEach(v => {

                const shift = state.shifts.find(s => s.id === v.shiftId);

                const shiftTitle = shift ? shift.title.replace(/,/g, "") : "Unknown";

                csvContent += `"${v.name}","${v.email}","${v.phone}","${shiftTitle}","${v.type}"\n`;

            });


            const encodedUri = encodeURI(csvContent);

            const link = document.createElement("a");

            link.setAttribute("href", encodedUri);

            link.setAttribute("download", "Temple_Cleaning_Volunteer_Roster_2026.csv");

            document.body.appendChild(link);

            link.click();

            document.body.removeChild(link);

        }
